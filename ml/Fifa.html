def encode_body_type(body_type):

        return 1
    elif body_type.strip() =="Lean":

def encode_stats(stats):
    if stats is np.nan:
        return np.nan
    else:
        only = stats.strip()
        numarr = only.split('+')
        total = int(numarr[0]) + int(numarr[1])
        return total
